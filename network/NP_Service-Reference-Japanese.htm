SceNpManagerSigninUpdateParam
 [構造体]
サインインチケット更新パラメータ

構造

#include <np_service.h>

typedef struct SceNpManagerSigninUpdateParam {
	SceSize size;
	SceLong64 ttl;
	int result;
	int reserved;
} SceNpManagerSigninUpdateParam;
メンバ

  size     自身の構造体のサイズ  
  ttl     更新したチケットの有効期間  
  result     更新結果  
  reserved     予約領域（0を指定してください）  

解説

サインインチケット更新のパラメータを表す構造体です。
sizeには自身の構造体のサイズを指定してください。ttlとresultにはサインインチケット更新の結果が返されます。reservedには必ず０を指定してください。

	// エラー処理
}

sceNpManagerSigninUpdateInitStart
 [関数]
サインインチケット更新処理の開始

形式

#include <np_service.h>

int sceNpManagerSigninUpdateInitStart(
	SceNpManagerSigninUpdateParam *param
);
呼出条件

割り込みハンドラから呼び出し不可
スレッドから呼び出し可能(割り込み許可状態で呼ぶこと)
マルチスレッドセーフでない

引数

  param     更新パラメータ  

返り値

正常終了の場合、0を返します。
エラーがあった場合、エラーコードを返します。

解説

サインインチケットの更新を行います。
本APIを呼び出す前に必ずNPサインインユーティリティを使ったサインイン処理を行ってください。（ユーティリティによるサインインを行う前に本APIを呼び出すとSCE_NP_MANAGER_ERROR_NO_CREDENTIALエラーとなります）

サインインチケットは、発行されてからTTL時間（現在24時間に設定されています）が過ぎると有効期限が切れたと判断され、サインインチケットを使用するAPIが正常に機能しなくなります。これを避けるためには、NPサインインユーティリティを再度呼び出してサインインしなおすか、あるいは本APIを使ってサインインチケットを更新してください。

param構造体のsizeには、param構造体自身のサイズを指定します。その他のメンバには、0を指定してください。呼び出しに成功すると、内部の状態がSCE_NP_MANAGER_SIGNIN_UPDATE_STATUS_NONEからSCE_NP_MANAGER_SIGNIN_UPDATE_STATUS_RUNNINGへと変わります。内部の状態を取得するには、sceNpManagerSigninUpdateGetStatus()を呼び出してください。チケット更新が成功・失敗に関わらず完了すると、内部状態がSCE_NP_MANAGER_SIGNIN_UPDATE_STATUS_FINISHEDになります。
SCE_NP_MANAGER_SIGNIN_UPDATE_STATUS_FINISHEDになると、param構造体のttlとresultにサインインチケット更新処理の結果の値が設定されます。エラーが発生した場合、ttlには0、resultにはエラーコードが設定されます。正常に終了した場合、ttlには更新したチケットのTTLが、resultには0が設定されます。
SCE_NP_MANAGER_SIGNIN_UPDATE_STATUS_FINISHEDになったら、sceNpManagerSigninUpdateShutdownStart()を呼び出して内部リソースを開放してください。
※本当のUtility APIとは異なり、sceNpManagerSigninUpdateShutdownStart()の呼び出しから返った時点で終了処理は完了しています。

使用例

int ret;
SceNpManagerSigninUpdateParam param;

memset(&param, 0x00, sizeof(param));

param.size = sizeof(param);
ret = sceNpManagerSigninUpdateInitStart(&param);
if (ret < 0){
	// エラー処理
}

// チケット更新処理の完了を待機
while(sceNpManagerSigninUpdateGetStatus() !=
	SCE_NP_MANAGER_SIGNIN_UPDATE_STATUS_FINISHED)
{
	sceKernelDelayThread(100 * 1000);
}

// 状態が SCE_NP_MANAGER_SIGNIN_UPDATE_STATUS_FINISHED で
// あれば sceNpManagerSigninUpdateShutdownStart() は必ず成功します。
// チケット更新処理がエラー終了した場合でも、終了処理は行ってください。
if (param.result < 0) {
	// エラー処理
	sceNpManagerSigninUpdateShutdownStart();
}

 [関数]
内部のメモリ使用状況を取得

形式

#include <np_service.h>

int sceNpServiceGetMemoryStat(
	struct SceNpMemoryStat *mstat
);
呼出条件

割り込みハンドラから呼び出し不可
スレッドから呼び出し可能(割り込み許可状態で呼ぶこと)
マルチスレッドセーフである

引数

  mstat     メモリ使用状況構造体へのポインタ  

返り値

正常終了の場合、0を返します。
エラーがあった場合、エラーコードを返します。

解説

NP service内部のメモリプールの使用状況を取得し、mstatに格納します。

使用例

struct SceNpMemoryStat stat;

ret = sceNpServiceGetMemoryStat(&stat);
if(ret < 0){
	/* エラー処理 */
}

printf("poolsize: %d, maxsize: %d, freesize: %d\n", stat.poolsize, stat.maxsize, stat.freesize);
