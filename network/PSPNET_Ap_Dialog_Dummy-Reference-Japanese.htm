SceNetApDialogDummyParam
 [構造体]
APネットワーク接続パラメータ構造体

構造

#include <pspnet_ap_dialog_dummy.h>

struct SceNetApDialogDummyParam {
	char ssid[SCE_UTILITY_NET_PARAM_SSID_LEN + 1];
	SceUChar8 padding1[1];
	unsigned int auth_proto;
	char wep_key[SCE_UTILITY_NET_PARAM_WEP_KEY_LEN];
	SceUChar8 padding2[3];
	unsigned int how_to_set_ip;
	union ip_opt {
		struct static_ip {
			char ip_address[SCE_UTILITY_NET_PARAM_IPV4_ADDR_STR_LEN];
			char netmask[SCE_UTILITY_NET_PARAM_IPV4_ADDR_STR_LEN];
			char default_route[SCE_UTILITY_NET_PARAM_IPV4_ADDR_STR_LEN];
		} static_ip;
		struct pppoe {
			char auth_name[SCE_UTILITY_NET_PARAM_AUTH_NAME_LEN];
			char auth_key[SCE_UTILITY_NET_PARAM_AUTH_KEY_LEN];
		} pppoe;
	} ip_opt;
	unsigned int dns_flag;
	char primary_dns[SCE_UTILITY_NET_PARAM_IPV4_ADDR_STR_LEN];
	char secondary_dns[SCE_UTILITY_NET_PARAM_IPV4_ADDR_STR_LEN];
	unsigned int http_proxy_flag;
	char http_proxy_server[SCE_UTILITY_NET_PARAM_HOSTNAME_LEN];
	SceUShort16 http_proxy_port;
	SceUChar8 padding3[2];
	SceUInt32 auth_8021x_type;
	union auth_8021x_opt {
		struct eap_md5 {
			SceUChar8 auth_name[SCE_UTILITY_NET_PARAM_AUTH_NAME_LEN];
			SceUChar8 auth_key[SCE_UTILITY_NET_PARAM_AUTH_KEY_LEN];
		} eap_md5;
	} auth_8021x_opt;
	SceUInt32 wpa_key_type;
	SceUChar8 wpa_key[SCE_UTILITY_NET_PARAM_WPA_KEY_LEN];
};
メンバ

  ssid     接続先のSSID  
  padding1     パディング  
  auth_proto     認証プロトコル  
  wep_key     WEPキー  
  padding2     パディング  
  how_to_set_ip     IPアドレス設定方法  
  ip_opt.static_ip.ip_address     IPアドレス(固定IP指定時に有効)。  
  ドット表現の文字列で指定します(例: "192.168.0.100")。  
  ip_opt.static_ip.netmask     ネットマスク(固定IP指定時に有効)。  
  ドット表現の文字列で指定します(例: "255.255.255.0")。  
  ip_opt.static_ip.default_route     デフォルト経路(固定IP指定時に有効)。  
  ドット表現の文字列で指定します(例: "192.168.0.1")。  
  ip_opt.pppoe.auth_name     PPPoEユーザ名(PPPoE指定時に有効)  
  ip_opt.pppoe.auth_key     PPPoEパスワード(PPPoE指定時に有効)  
  dns_flag     DNS指定フラグ  
  primary_dns     プライマリDNS。  
  ドット表現の文字列で指定します(例: "192.168.0.1")。  
  secondary_dns     セカンダリDNS。  
  ドット表現の文字列で指定します(例: "192.168.0.2")。  
  http_proxy_flag     HTTPプロキシ使用フラグ  
  http_proxy_server     HTTPプロキシサーバー  
  http_proxy_port     HTTPプロキシポート番号  
  padding3     パディング  
  auth_8021x_type     802.1X認証タイプ  
  auth_8021x_opt.eap_md5.auth_name     802.1XユーザーID  
  (802.1X認証選択時に有効)  
  auth_8021x_opt.eap_md5.auth_key     802.1Xパスワード  
  (802.1X認証選択時に有効)  
  wpa_key_type     WPAキータイプ  
  wpa_key     WPAキー  

解説

APネットワーク接続時に、接続先のAP情報を指定するための構造体です。
auth_proto、how_to_set_ip、dns_flag、http_proxy_flag、auth_8021x_type、wpa_key_typeに指定する値はユーティリティのネットワーク設定ドキュメントを参照してください。

sceNetApDialogDummyConnect
 [関数]
APネットワークへ接続

形式

#include <pspnet_ap_dialog_dummy.h>

int sceNetApDialogDummyConnect(
	struct SceNetApDialogDummyParam *param
);
呼出条件

割り込みハンドラから呼び出し不可
スレッドから呼び出し可能(割り込み許可状態で呼ぶこと)
マルチスレッドセーフでない

引数

  param     APネットワーク接続パラメータ構造体へのポインタ  

返り値

正常終了の場合、0を返します。
エラーがあった場合、以下のいずれかのエラーコードを返します。
SCE_ERROR_BUSY
SCE_ERROR_INVALID_POINTER
pspnet_apctlの内部エラー(0x80410a80〜0x80410aff)
libnetcnfの内部エラー(0x80410980〜0x804109ff)

解説

paramで与えられたパラメータを基に、APネットワーク接続を試みます。
本関数呼び出し後は、sceNetApDialogDummyGetState()によって接続処理の終了を待ちます。

使用例

#define TARGET_SSID "foobar"
struct SceNetApDialogDummyParam param;

memset(&param, 0, sizeof(param));
strcpy(param.ssid, TARGET_SSID);

ret = sceNetApDialogDummyConnect(&param);
if(ret < 0){
	/* エラー処理 */
}

sceNetApDialogDummyGetState()

sceNetApDialogDummyGetState
 [関数]
APネットワーク接続状態取得関数

形式

#include <pspnet_ap_dialog_dummy.h>

int sceNetApDialogDummyGetState(
	struct SceNetApDialogDummyStateInfo *stat_info
);
呼出条件

割り込みハンドラから呼び出し不可
スレッドから呼び出し可能(割り込み許可状態で呼ぶこと)
マルチスレッドセーフでない

引数

  stat_info     APネットワーク接続状態構造体へのポインタ  

返り値

正常終了の場合、0を返します。
エラーがあった場合、以下のエラーコードを返します。
SCE_ERROR_INVALID_POINTER

解説

APネットワーク接続処理の状態を取得します。

使用例

struct SceNetApDialogDummyStateInfo stat_info;

ret = sceNetApDialogDummyConnect(&param);
if(ret < 0){
	/* エラー処理 */
}

while(1){
	ret = sceNetApDialogDummyGetState(&stat_info);
	if(ret == 0){
		/* 接続完了もしくはエラーによる切断完了を待つ */
		if(stat_info.state == SceNetApDialogDummyState_Connected ||
		stat_info.state == SceNetApDialogDummyState_Disconnected)
		break;
	}
	/* 適当なポーリング間隔をあける */
	sceKernelDelayThread(AP_DIALOG_DUMMY_WAIT_TIME);
}
