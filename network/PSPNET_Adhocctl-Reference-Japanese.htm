pspnet_adhocctl


SceNetAdhocctlAdhocId
 [構造体]
アドホックID構造体

構造

#include <pspnet_adhocctl.h>

#define SCE_NET_ADHOCCTL_ADHOCID_LEN		9
#define SCE_NET_ADHOCCTL_ADHOCTYPE_PRODUCT_ID	0
#define SCE_NET_ADHOCCTL_ADHOCTYPE_RESERVED	1
#define SCE_NET_ADHOCCTL_ADHOCTYPE_SYSTEM	2
struct SceNetAdhocctlAdhocId {
	int type;
	SceChar8 data[SCE_NET_ADHOCCTL_ADHOCID_LEN];
	SceUChar8 padding[3];
};
メンバ

  data     アドホックIDデータ  
  type     アドホックIDタイプ  
  padding     パディングデータ  

解説

アドホックIDを指定する構造体です。ライブラリ初期化関数sceNetAdhocctlInit()の引数に指定します。
dataには、原則的にタイトル品番（アルファベット4文字＋数字5文字）を指定してください。
なお異なるタイトルアプリケーション間での通信など、タイトル品番以外の指定を希望される場合については、PSP®ネットワークライブラリを参照してください。
typeは、アドホックIDの属性を表します。
typeには、以下のいずれかの値を指定してください。

  値     説明  
  SCE_NET_ADHOCCTL_ADHOCTYPE_PRODUCT_ID     製品版アドホックID  
  SCE_NET_ADHOCCTL_ADHOCTYPE_RESERVED     デバッグ用アドホックID  
  SCE_NET_ADHOCCTL_ADHOCTYPE_SYSTEM     PSP®システム用アドホックID  


SCE_NET_ADHOCCTL_ADHOCTYPE_SYSTEMは、PSP®システム用のアドホックIDです。

sceNetAdhocctlInit
 [関数]
アドホックネットワーク接続ライブラリ初期化

形式

#include <pspnet_adhocctl.h>

int sceNetAdhocctlInit(
	int stacksize,
	int prio,
	const struct SceNetAdhocctlAdhocId *adhoc_id
);
呼出条件

割り込みハンドラから呼び出し不可
スレッドから呼び出し可能(割り込み許可状態で呼ぶこと)
マルチスレッドセーフでない

引数

  stacksize     内部スレッドのスタックサイズ  
  prio     内部スレッドの優先度  
  adhoc_id     アドホックID  

返り値

正常終了の場合、0を返します。
エラーがあった場合、以下のいずれかのエラーコードを返します。
SCE_ERROR_NET_ADHOCCTL_ALREADY_INITIALIZED
SCE_ERROR_NET_ADHOCCTL_STACKSIZE_TOO_SHORT
SCE_ERROR_NET_ADHOCCTL_INVALID_ARG
pspnetの内部エラー(0x80410180〜0x804101ff)

解説

インタフェースの立ち上げとアドホックネットワーク接続ライブラリの初期化を行います。
内部スレッドを作成し、スタートします。
adhoc_idには、割り当てられたアドホックIDを指定してください。

使用例

#include <pspnet_adhocctl.h>

#define ADHOCCTL_HADNLER_STACK_SIZE	(1 * 1024)	/* 1KB */
#define ADHOCCTL_PRIO		30
int ret;
struct SceNetAdhocctlAdhocId adhoc_id;

adhoc_id.type = SCE_NET_ADHOCCTL_ADHOCTYPE_PRODUCT_ID;
memcpy(adhoc_id.data, "123456789", SCE_NET_ADHOCCTL_ADHOCID_LEN);

ret = sceNetAdhocctlInit(
		SCE_NET_PSPNET_ADHOCCTL_LEAST_STACK_SIZE
			+ ADHOCCTL_HANDLER_STACK_SIZE,
		ADHOCCTL_PRIO, &adhoc_id);
if(ret < 0){
	// エラー処理
}
備考

内部スレッドのスタックサイズには、SCE_NET_PSPNET_ADHOCCTL_LEAST_STACK_SIZE（=3072バイト）とsceNetAdhocctlAddHandler()で指定するハンドラ関数で使用するスタックサイズの合計を指定してください。

sceNetAdhocctlConnect
 [関数]
アドホック無線ネットワーク接続関数

形式

#include <pspnet_adhocctl.h>

int sceNetAdhocctlConnect(
	const struct SceNetAdhocctlGroupName *group_name
);
呼出条件

割り込みハンドラから呼び出し不可
スレッドから呼び出し可能(割り込み許可状態で呼ぶこと)
マルチスレッドセーフでない

引数

  group_name     グループ名  

返り値

正常終了の場合、0を返します
エラーがあった場合、以下のいずれかのエラーコードを返します。
SCE_ERROR_NET_ADHOCCTL_NOT_INITIALIZED
SCE_ERROR_NET_ADHOCCTL_INVALID_ARG
SCE_ERROR_NET_ADHOCCTL_BUSY

解説

本関数は、SCE_NET_ADHOCCTL_STATE_DISCONNECTED状態でのみ有効です。
本関数は、内部スレッドへイベントを通知するためのノンブロッキング関数です。
内部スレッドはイベントを受け取ると、group_nameで指定されたアドホック無線ネットワーク（IBSS）へ接続します。
接続完了や、エラーなどのイベントは、イベントハンドラに通知されます。
接続が完了すると、イベントハンドラにSCE_NET_ADHOCCTL_EVENT_CONNECTが渡され、状態がSCE_NET_ADHOCCTL_STATE_CONNECTEDになります。

group_nameの仕様は、次の通りです。
  • 英数字のみ使用可能 ('0' 〜 '9')、('A' 〜 'Z')、('a' 〜 'z')
  • NULL文字は特に必要ありませんが、途中にNULL文字があればそこを終端として扱います。
    1バイト目にNULL文字、あるいはgroup_name自体にNULLを指定した場合、group_nameが指定されなかったと判断してSceNetAdhocctlAdhocIdのみでSSIDを構成します。

使用例

int ret;
struct SceNetAdhocctlGroupName group_name;

memset(&group_name, 0x00, sizeof(struct SceNetAdhocctlGroupName));
memcpy(group_name.data, "test", strlen("test"));
ret = sceNetAdhocctlConnect(&group_name);
if (ret < 0) {
	// エラー処理
}
wait_connect();   /* SCE_NET_ADHOCCTL_EVENT_CONNECT を待つ */
備考

本関数の代替手段として、sceNetAdhocctlCreate()およびsceNetAdhocctlJoin()を利用することもできます。それぞれの特徴については、「ネットワーク概要ドキュメント」を参照してください。

備考

同一のグループ名で複数のPSP®端末が本関数を使用した場合、それぞれBSSIDの異なるIBSSが作成されますが、各端末がビーコンの送信を譲り合う症状が確認されています。
sceNetAdhocctlScan()は、ビーコンを調べてIBSSを検出しますので、複数台の端末が存在していてもそのうちの１台のみ検出することになります。
ビーコンを譲り合う現象はハードウェアの仕様ですので、これをソフト的に回避する方法はございません。
同一のグループ名のIBSSを複数作成する場合は、sceNetAdhocctlConnect()を利用してください。

sceNetAdhocctlJoin
 [関数]
アドホック無線ネットワーク参加関数

形式

#include <pspnet_adhocctl.h>

int sceNetAdhocctlJoin(
	const struct SceNetAdhocctlScanInfo *scan_info
);
呼出条件

割り込みハンドラから呼び出し不可
スレッドから呼び出し可能(割り込み許可状態で呼ぶこと)
マルチスレッドセーフでない

引数

  scan_info     スキャン情報  

返り値

正常終了の場合、0を返します
エラーがあった場合、以下のいずれかのエラーコードを返します。
SCE_ERROR_NET_ADHOCCTL_NOT_INITIALIZED
SCE_ERROR_NET_ADHOCCTL_INVALID_ARG
SCE_ERROR_NET_ADHOCCTL_BUSY

解説

本関数は、SCE_NET_ADHOCCTL_STATE_DISCONNECTED状態でのみ有効です。
本関数は、内部スレッドへイベントを通知するためのノンブロッキング関数です。
内部スレッドはイベントを受け取ると、scan_infoで指定されたアドホック無線ネットワーク（IBSS）へ参加します。
接続完了や、エラーなどのイベントは、イベントハンドラに通知されます。
接続が完了すると、イベントハンドラにSCE_NET_ADHOCCTL_EVENT_CONNECTが渡され、状態がSCE_NET_ADHOCCTL_STATE_CONNECTEDになります。

本関数を使用する場合、前もってsceNetAdhocctlScan()の処理を終了させておく必要があります。スキャンの結果をもとに、参加するIBSSを選択してください。sceNetAdhocctlScan()を呼ばずに直接パラメータを設定しても、IBSSに参加することはできません。

使用例

int ret;
struct SceNetAdhocctlScanInfo *buf, *cur;
int buflen;
struct SceNetAdhocctlGroupName group_name; /* 参加したいグループ名 */

/* 既に作成されている IBSS をスキャンする */
ret = sceNetAdhocctlScan();
if (ret < 0) {
    // エラー処理
}
ret = wait_scan();   /* SCE_NET_ADHOCCTL_EVENT_SCAN を待つ */
if (ret < 0) {
    // エラー処理
}

ret = sceNetAdhocctlGetScanInfo(&buflen, NULL);
if (ret < 0) {
    // エラー処理
}
if (buflen == 0) {
    // 周囲に IBSS が存在していない
}

buf = (struct SceNetAdhocctlScanInfo *)malloc(buflen);
if (buf == NULL ) {
    // エラー処理
}

ret = sceNetAdhocctlGetScanInfo(&buflen, buf);
if (ret < 0) {
    // エラー処理
}
if (buflen == 0) {
    // エラー処理
}


/* スキャン結果から、グループ名の比較を行って参加するべきIBSSを探す。*/
/* 注意) スキャン結果には、同一のグループ名を持つIBSSが複数含まれている可能性
         があります。これらのIBSSは、BSSIDによって判別されています。*/
for (cur=buf; cur!=NULL; cur=cur->next) {
	if (0==strncmp(cur->group_name.data, group_name.data,
		sizeof(group_name.data)))
            break;
}

/* 目当てのIBSSはスキャン結果に含まれていなかった。 */
if (cur == NULL) {
    // エラー処理
}

ret = sceNetAdhocctlJoin(cur);
if (ret < 0) {
    // エラー処理
}
ret = wait_connect();
if (ret < 0) {
    // エラー処理
}
