SceNetApctlInfo
 [Unions]
AP network connection information union

Structure

#include <pspnet_apctl.h>

union SceNetApctlInfo {
	SceChar8 cnf_name[SCE_UTILITY_NET_PARAM_CNF_NAME_LEN];
	SceUChar8 bssid[SCE_NET_APCTL_BSSID_LEN];
	SceUChar8 ssid[SCE_NET_APCTL_SSID_LEN];
	SceSize ssidlen;
	SceUInt32 auth_proto;
	SceUChar8 rssi;
	SceUChar8 channel;
	SceUChar8 pwrsave;
	SceChar8 ip_address[SCE_UTILITY_NET_PARAM_IPV4_ADDR_STR_LEN];
	SceChar8 netmask[SCE_UTILITY_NET_PARAM_IPV4_ADDR_STR_LEN];
	SceChar8 default_route[SCE_UTILITY_NET_PARAM_IPV4_ADDR_STR_LEN];
	SceChar8 primary_dns[SCE_UTILITY_NET_PARAM_IPV4_ADDR_STR_LEN];
	SceChar8 secondary_dns[SCE_UTILITY_NET_PARAM_IPV4_ADDR_STR_LEN];
	SceUInt32 http_proxy_flag;
	SceChar8 http_proxy_server[SCE_UTILITY_NET_PARAM_HOSTNAME_LEN];
	SceUShort16 http_proxy_port;
	SceUInt32 auth_8021x_type;
	SceUInt32 browser_flag;
	SceUInt32 wifisvc_config;
};
Members

  cnf_name     Connection name specified during connection. String.  
  bssid     BSSID of destination AP. 6-byte binary format.  
  ssid     SSID of destination AP. Length can be obtained in ssidlen.  
  ssidlen     SSID length  
  auth_proto     Security  
  rssi     Electromagnetic field intensity  
  channel     Channel  
  pwrsave     Power saving mode  
  ip_address     IP address. String.  
  netmask     Subnet mask. String.  
  default_route     Default route. String.  
  primary_dns     Primary DNS. String.  
  secondary_dns     Secondary DNS. String.  
  http_proxy_flag     Proxy server specification flag.  
  http_proxy_server     Proxy server. String.  
  http_proxy_port     Proxy server port number.   
  auth_8021x_type     802.1X authentification type  
  browser_flag     Browser start-up flag  
  wifisvc_config     Wireless hotspot configuration  

Description

This union is used for obtaining AP network connection information.

The values for auth_proto are as follows.
  Macro     Description  
  SCE_NET_APCTL_INFO_AUTH_PROTO_NOAUTH     No authentication  
  SCE_NET_APCTL_INFO_AUTH_PROTO_WEP     WEP  


The values for http_proxy_flag are as follows.
  Macro     Description  
  SCE_NET_APCTL_INFO_HTTP_PROXY_OFF     HTTP proxy not in effect  
  SCE_NET_APCTL_INFO_HTTP_PROXY_ON     HTTP proxy in effect  


The values for auth_8021x_type are as follows.
  Macro     Description  
  SCE_NET_APCTL_INFO_8021X_NONE     No 802.1X authentication  
  SCE_NET_APCTL_INFO_8021X_EAP_MD5     EAP-MD5  


The values for browser_flag are as follows.
  Macro     Description  
  SCE_NET_APCTL_INFO_BROWSER_OFF     Browser disabled  
  SCE_NET_APCTL_INFO_BROWSER_ON     Browser enabled  


The values for wifisvc_config are as follows.
  Macro     Description  
  SCE_NET_APCTL_INFO_WIFISVC_CONFIG_NONE     Not a wireless hotspot configuration  
  SCE_NET_APCTL_INFO_WIFISVC_CONFIG_TMOBILE     T-Mobile-specific configuration

sceNetApctlInit
 [Function]
Initialize the AP Network Connection library

Syntax

#include <pspnet_apctl.h>

int sceNetApctlInit(
	int stacksize,
	int prio
);
Calling Conditions

Cannot be called from an interrupt handler
Can be called from a thread (must be in interrupt-enabled state)
Not multithread safe

Arguments

  stacksize     Stack size of internal thread  
  prio     Priority level of internal thread  

Return value

If this function completes successfully, 0 is returned.
If an error occurs, one of the following error codes is returned.
SCE_ERROR_NET_APCTL_NOT_TERMINATED

Description

This function initializes the AP Network Connection library.
This function creates an internal thread according to stacksize and prio.

Example

#define SCE_APCTL_STACKSIZE 0x8000
#define SCE_APCTL_PRIO 33

ret = sceNetApctlInit(SCE_APCTL_STACKSIZE, SCE_APCTL_PRIO);
if(ret < 0){
	/* Error handling */
}
See Also

sceNetApctlTerm()

apctlhandler
 [Function]
Event handler prototype

Syntax

#include <pspnet_apctl.h>

void apctlhandler(
	int prev_state,
	int new_state,
	int event,
	int error_code,
	void *arg
);
Arguments

  prev_state     State before transition  
  new_state     State after transition  
  event     Event which caused the state transition  
  error_code     Error code  
  arg     arg previously specified by sceNetApctlAddHandler()  


Description

This event handler is called from an internal thread when a state transition occurs. The gp value is set to the gp value of the module where the event handler is defined.
When the event is SCE_NET_APCTL_EVENT_ERROR, the error code is passed in error_code.

Example

void
wlanapctl_handler(
	int prev_state,
	int new_state,
	int event,
	int error_code,
	void *arg
	)
{
	printf("event: %d -- %d(0x%x) --> %d\n", prev_state, event,
            error_code, new_state);
}
Notes

Make sure that the event handler and a thread that calls an AP network connection library API function will not both be waiting for the same resource (such as sceKernelWaitSema() waiting for the same semaphore). Otherwise, the thread that called the AP network connection library API function and the internal thread of the AP network connection library may get deadlocked.
