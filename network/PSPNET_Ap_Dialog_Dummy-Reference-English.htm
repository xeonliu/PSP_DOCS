		} static_ip;
		struct pppoe {
			char auth_name[SCE_UTILITY_NET_PARAM_AUTH_NAME_LEN];

	char primary_dns[SCE_UTILITY_NET_PARAM_IPV4_ADDR_STR_LEN];
	char secondary_dns[SCE_UTILITY_NET_PARAM_IPV4_ADDR_STR_LEN];

	char http_proxy_server[SCE_UTILITY_NET_PARAM_HOSTNAME_LEN];

	SceUInt32 wpa_key_type;

};

sceNetApDialogDummyConnect
 [Function]
Connect to AP network

Syntax

#include <pspnet_ap_dialog_dummy.h>

int sceNetApDialogDummyConnect(
	struct SceNetApDialogDummyParam *param
);
Calling Conditions

Cannot be called from an interrupt handler.
Can be called from a thread (must be called in interrupt-enabled state).
Not multithread safe.

Arguments

  param     Pointer to AP network connection parameter structure  

Return Value

Returns 0 on normal completion.
If an error occurs, one of the following error codes is returned.
SCE_ERROR_BUSY
SCE_ERROR_INVALID_POINTER
pspnet_apctl internal error (0x80410a80 - 0x80410aff)
libnetcnf internal error (0x80410980 - 0x804109ff)

Description

This function attempts to make an AP network connection based on the parameters provided by param.
After calling this function, use sceNetApDialogDummyGetState() to wait for the connection to be completed.

Example

#define TARGET_SSID "foobar"
struct SceNetApDialogDummyParam param;

memset(&param, 0, sizeof(param));
strcpy(param.ssid, TARGET_SSID);

ret = sceNetApDialogDummyConnect(&param);
if(ret < 0){
	/* Error handling  */
}

sceNetApDialogDummyGetState()

sceNetApDialogDummyGetState
 [Function]
Get AP network connection state

Syntax

#include <pspnet_ap_dialog_dummy.h>

int sceNetApDialogDummyGetState(
	struct SceNetApDialogDummyStateInfo *stat_info
);
Calling Conditions

Cannot be called from an interrupt handler.
Can be called from a thread (must be called in interrupt-enabled state).
Not multithread safe.

Arguments

  stat_info     Pointer to AP network connection state structure  

Return Value

Returns 0 on normal completion.
If an error occurs, one of the following error codes is returned.
SCE_ERROR_INVALID_POINTER

Description

This function gets the state of an AP network connection.

Example

struct SceNetApDialogDummyStateInfo stat_info;

ret = sceNetApDialogDummyConnect(&param);
if(ret < 0){
	/* Error handling  */
}

while(1){
	ret = sceNetApDialogDummyGetState(&stat_info);
	if(ret == 0){
		/* Wait for connected, or for disconnected due to error */
		if(stat_info.state == SceNetApDialogDummyState_Connected ||
		stat_info.state == SceNetApDialogDummyState_Disconnected)
		break;
	}
	/* Pause for suitable polling interval  */
	sceKernelDelayThread(AP_DIALOG_DUMMY_WAIT_TIME);
}
