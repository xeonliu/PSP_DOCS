sceNetResolverCreate
 [Function]
Create resolver context

Syntax

#include <pspnet_resolver.h>

int sceNetResolverCreate(
	int *rid,
	void *buf,
	SceSize buflen
);
Calling Conditions

Cannot be called from an interrupt handler
Can be called from a thread (must be called in an interrupt-enabled state)
Multithread safe

Arguments

  rid     Pointer to area for storing resolver ID that is obtained  
  buf     Pointer to send/receive buffer that is used by the created context  
  buflen     Size of send/receive buffer   

Return Value

Returns 0 on normal completion.
If an error occurs, one of the following error codes is returned.
SCE_ERROR_NET_RESOLVER_INVALID_PTR
SCE_ERROR_NET_RESOLVER_INVALID_BUFLEN
SCE_ERROR_NET_RESOLVER_NO_MEM
SCE_ERROR_NET_RESOLVER_ID_MAX

Description

This function creates a resolver context and stores the resolver ID in the area specified by rid.
It obtains the area used to manage the context from the memory pool that was allocated by sceNetResolverInit().
This area is freed by calling sceNetResolverDelete() and specifying the resolver ID which was obtained.

Example

char buf[SCE_PSPNET_RESOLVER_NTOA_BUFSIZE_MAX];
int rid;

ret = sceNetResolverCreate(&rid, buf, sizeof(buf));
if(ret < 0){
	/* Error handling */
}

sceNetResolverStartNtoA
 [Function]
Start lookup name resolution

Syntax

#include <pspnet_resolver.h>
#include <pspnet/netinet/in.h>

int sceNetResolverStartNtoA(
	int rid,
	const char *hostname,
	struct SceNetInetInAddr *addr,
	SceUInt32 timeout,
	int retry
);
Calling Conditions

Cannot be called from an interrupt handler
Can be called from a thread (must be called in an interrupt-enabled state)
Multithread safe

Arguments

  rid     Resolver ID  
  hostname     Host name on which to perform name resolution  
  addr     Pointer to area for storing IP address corresponding to host name  
  timeout     Inquiry retry interval  
  retry     Inquiry retry count  

Return Value

Returns 0 on normal completion.
If an error occurs, one of the following error codes is returned.
SCE_ERROR_NET_RESOLVER_INVALID_ID
SCE_ERROR_NET_RESOLVER_INVALID_PTR
SCE_ERROR_NET_RESOLVER_ID_NOT_FOUND
SCE_ERROR_NET_RESOLVER_CTX_BUSY
SCE_ERROR_NET_RESOLVER_NO_DNS_SERVER
SCE_ERROR_NET_RESOLVER_BUF_NO_SPACE
SCE_ERROR_NET_RESOLVER_INTERNAL
SCE_ERROR_NET_RESOLVER_STOPPED
SCE_ERROR_NET_RESOLVER_TIMEOUT
SCE_ERROR_NET_RESOLVER_SOCKET
SCE_ERROR_NET_RESOLVER_INVALID_PACKET
SCE_ERROR_NET_RESOLVER_NOT_SUPPORTED
SCE_ERROR_NET_RESOLVER_RES_PACKET_FORMAT
SCE_ERROR_NET_RESOLVER_RES_SERVER_FAILURE
SCE_ERROR_NET_RESOLVER_NO_HOST
SCE_ERROR_NET_RESOLVER_RES_NOT_IMPLEMENTED
SCE_ERROR_NET_RESOLVER_RES_SERVER_REFUSED
SCE_ERROR_NET_RESOLVER_NO_RECORD

Description

This function starts lookup name resolution using the resolver context specified by rid.
It finds the IP address corresponding to the host name specified by hostname and stores it in the area specified by addr.
Name resolution is performed by the calling thread, and this function blocks until name resolution completes normally or an error occurs.

Example

#define SERVER_NAME "www.scei.co.jp"
#define RESOLVER_TIMEOUT (5 * 1000 * 1000)
#define RESOLVER_RETRY 5

struct SceNetInetInAddr addr;

ret = sceNetResolverStartNtoA(rid, SERVER_NAME, &addr, RESOLVER_TIMEOUT, RESOLVER_RETRY);
if(ret < 0){
	/* Error handling */
}

sceNetResolverStartAtoN
 [Function]
Start reverse lookup name resolution

Syntax

#include <pspnet_resolver.h>
#include <pspnet/netinet/in.h>

int sceNetResolverStartAtoN(
	int rid,
	const struct SceNetInetInAddr *addr,
	char *hostname,
	SceSize hostname_len,
	SceUInt32 timeout,
	int retry
);
Calling Conditions

Cannot be called from an interrupt handler
Can be called from a thread (must be called in an interrupt-enabled state)
Multithread safe

Arguments

  rid     Resolver ID  
  addr     Pointer to area for storing IP address used for reverse lookup  
  hostname     Pointer to area for storing host name corresponding to addr  
  hostname_len     Length of hostname area  
  timeout     Inquiry retry interval  
  retry     Inquiry retry count  

Return Value

Returns 0 on normal completion.
If an error occurs, one of the following error codes is returned.
SCE_ERROR_NET_RESOLVER_INVALID_ID
SCE_ERROR_NET_RESOLVER_INVALID_PTR
SCE_ERROR_NET_RESOLVER_ID_NOT_FOUND
SCE_ERROR_NET_RESOLVER_CTX_BUSY
SCE_ERROR_NET_RESOLVER_NO_DNS_SERVER
SCE_ERROR_NET_RESOLVER_BUF_NO_SPACE
SCE_ERROR_NET_RESOLVER_INTERNAL
SCE_ERROR_NET_RESOLVER_STOPPED
SCE_ERROR_NET_RESOLVER_SOCKET
SCE_ERROR_NET_RESOLVER_TIMEOUT
SCE_ERROR_NET_RESOLVER_INVALID_PACKET
SCE_ERROR_NET_RESOLVER_NOT_SUPPORTED
SCE_ERROR_NET_RESOLVER_RES_PACKET_FORMAT
SCE_ERROR_NET_RESOLVER_RES_SERVER_FAILURE
SCE_ERROR_NET_RESOLVER_NO_HOST
SCE_ERROR_NET_RESOLVER_RES_NOT_IMPLEMENTED
SCE_ERROR_NET_RESOLVER_RES_SERVER_REFUSED
SCE_ERROR_NET_RESOLVER_NO_RECORD

Description

This function starts reverse lookup name resolution using the resolver context specified by rid.
It finds the host name corresponding to the IP address specified by addr and stores the host name in the area indicated by hostname/hostname_len.
If hostname_len is insufficient for the host name that was returned, only the first hostname_len bytes of the name are stored.

Name resolution is performed by the calling thread, and this function blocks until name resolution completes normally or an error occurs.

Example

#define RESOLVER_TIMEOUT (5 * 1000 * 1000)
#define RESOLVER_RETRY 5

char hostname[256];
struct SceNetInetInAddr addr = XXX; /* Assume that IP address has been placed in addr */

ret = sceNetResolverStartAtoN(rid, &addr, hostname, sizeof(hostname),
RESOLVER_TIMEOUT, RESOLVER_RETRY);
if(ret < 0){
	/* Error handling */
}

sceNetResolverPollAsync
 [Function]
Get status of non-blocking name resolution

Syntax

#include <pspnet_resolver.h>
#include <pspnet/netinet/in.h>

int sceNetResolverPollAsync(
	int rid,
	void *result,
);
Calling Conditions

Cannot be called from an interrupt handler
Can be called from a thread (must be called in an interrupt-enabled state)
Multithread safe

Arguments

  rid     Resolver ID  
  result     Unused  

Return Value

1 is returned if name resolution is still in progress. Otherwise, 0 is returned.
If an error occurs, one of the following error codes is returned.
SCE_ERROR_NET_RESOLVER_ID_NOT_FOUND
SCE_ERROR_NET_RESOLVER_INVALID_MODE
SCE_ERROR_NET_RESOLVER_NOT_STARTED
SCE_ERROR_NET_RESOLVER_NO_DNS_SERVER
SCE_ERROR_NET_RESOLVER_BUF_NO_SPACE
SCE_ERROR_NET_RESOLVER_INTERNAL
SCE_ERROR_NET_RESOLVER_STOPPED
SCE_ERROR_NET_RESOLVER_TIMEOUT
SCE_ERROR_NET_RESOLVER_SOCKET
SCE_ERROR_NET_RESOLVER_INVALID_PACKET
SCE_ERROR_NET_RESOLVER_NOT_SUPPORTED
SCE_ERROR_NET_RESOLVER_RES_PACKET_FORMAT
SCE_ERROR_NET_RESOLVER_RES_SERVER_FAILURE
SCE_ERROR_NET_RESOLVER_NO_HOST
SCE_ERROR_NET_RESOLVER_RES_NOT_IMPLEMENTED
SCE_ERROR_NET_RESOLVER_RES_SERVER_REFUSED
SCE_ERROR_NET_RESOLVER_NO_RECORD

Description

This function gets the status of non-blocking name resolution that was previously started for the resolver context specified by rid. If this function is called before sceNetResolverStartNtoAAsync() or sceNetResolverStartAtoNAsync(), an SCE_ERROR_NET_RESOLVER_INVALID_MODE error will be returned.
This function is non-blocking.

Example

#define SERVER_NAME "www.scei.co.jp"
#define RESOLVER_TIMEOUT (5 * 1000 * 1000)
#define RESOLVER_RETRY 5
#define DELAY_TIME (1000 * 1000)

struct SceNetInetInAddr addr;

/* Start non-blocking name resolution */
ret = sceNetResolverStartNtoAAsync(rid, SERVER_NAME, &addr, RESOLVER_TIMEOUT, RESOLVER_RETRY);
if(ret < 0){
	/* Error handling*/
}

/* Loop waiting for completion of name resolution */
while(1){
	ret = sceNetResolverPollAsync(rid, NULL);
	if(ret < 0){
		/* Error handling*/
	}
	else if(ret == 0){
		/* Completion of name resolution */
		break;
	}
	sceKernelDelayThread(DELAY_TIME);
}

/* Name resolution is completed at this time */
See Also

sceNetResolverStartAtoNAsync(), sceNetResolverStartNtoAAsync(), sceNetResolverStop(), sceNetResolverWaitAsync()
